    ```csharp
            Port = 161,
            CommunityString = $"test.{name}",
            MetricPolls = new List<MetricPollOptions>()
        };
        for (int i = 0; i < pollCount; i++)
        {
            device.MetricPolls.Add(new MetricPollOptions
            {
                IntervalSeconds = intervalSeconds,
                Oids = new List<string> { "1.3.6.1.2.1.1.1.0" }
            });
        }
        return device;
    }
    ```

       - Assert: `scheduler.Received(1).DeleteJob(Arg.Is<JobKey>(k => k.Name == "metric-poll-DEV-01-0"), Arg.Any<CancellationToken>())`
       - Assert: `intervalRegistry.Received(1).Unregister("metric-poll-DEV-01-0")`
       - Assert: `livenessVector.Received(1).Remove("metric-poll-DEV-01-0")`

    3. `ReconcileAsync_WithChangedInterval_ReschedulesJob`:
       - Setup: scheduler.GetJobKeys returns `{ JobKey("metric-poll-DEV-01-0") }`.
       - Setup: `intervalRegistry.TryGetInterval("metric-poll-DEV-01-0", out Arg.Any<int>()).Returns(x => { x[1] = 10; return true; });`
       - Call: `sut.ReconcileAsync(new[] { MakeDevice("DEV-01", 30) }, CancellationToken.None)` (interval changed from 10 to 30)
       - Assert: `scheduler.Received(1).RescheduleJob(Arg.Any<TriggerKey>(), Arg.Any<ITrigger>(), Arg.Any<CancellationToken>())`
       - Assert: `intervalRegistry.Received(1).Register(Arg.Is<string>(k => k.Contains("DEV-01")), 30)` (updated interval)

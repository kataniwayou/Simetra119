    /// <summary>
    /// Registers the ConfigMap watcher and dynamic poll scheduler for live config reload.
    /// In K8s: watches the simetra-config ConfigMap via K8s API for instant reload.
    /// Local dev: loads simetra-config.json from CONFIG_DIRECTORY once at startup (no hot-reload).
    /// Must be called after AddScheduling (depends on ISchedulerFactory, IJobIntervalRegistry).
    /// </summary>
    public static IServiceCollection AddConfigMapWatcher(this IServiceCollection services)
    {
        // DynamicPollScheduler is always registered (used by both K8s watcher and local startup)
        services.AddSingleton<IDynamicPollScheduler, DynamicPollScheduler>();

        if (KubernetesClientConfiguration.IsInCluster())
        {
            // K8s: ConfigMapWatcherService watches for live changes
            // Singleton-forward pattern (same as K8sLeaseElection)
            services.AddSingleton<ConfigMapWatcherService>();
            services.AddHostedService(sp => sp.GetRequiredService<ConfigMapWatcherService>());
        }

        return services;
    }
    ```

       // CFG-03: Local dev config loading from unified simetra-config.json
       if (!KubernetesClientConfiguration.IsInCluster())
       {
           var configDir = Environment.GetEnvironmentVariable("CONFIG_DIRECTORY")
               ?? Path.Combine(builder.Environment.ContentRootPath, "config");
           var configPath = Path.Combine(configDir, "simetra-config.json");

           if (File.Exists(configPath))
           {
               var jsonContent = await File.ReadAllTextAsync(configPath);
               var simetraConfig = JsoncParser.Parse(jsonContent);

               // Set Source on config-loaded metric polls (same as PostConfigure)
               foreach (var device in simetraConfig.Devices)
                   foreach (var poll in device.MetricPolls)
                       poll.Source = MetricPollSource.Configuration;

               var deviceRegistry = app.Services.GetRequiredService<IDeviceRegistry>();
               await deviceRegistry.ReloadAsync(simetraConfig.Devices);

               var pollRegistry = app.Services.GetRequiredService<IPollDefinitionRegistry>();
               pollRegistry.Reload(simetraConfig.Devices);

               var pollScheduler = app.Services.GetRequiredService<IDynamicPollScheduler>();
               await pollScheduler.ReconcileJobsAsync(simetraConfig.Devices, CancellationToken.None);

               var logger = app.Services.GetRequiredService<ILoggerFactory>()
                   .CreateLogger("Simetra.Startup");
               logger.LogInformation(
                   "Local config loaded from {Path}: {DeviceCount} devices",
                   configPath, simetraConfig.Devices.Count);
           }
       }
       ```

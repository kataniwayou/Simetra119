    Each scenario file should follow this structure:
    ```bash

    SCENARIO_NAME="MVC-XX: snmp_gauge [TYPE] exact value ([resolved_name])"

    RESPONSE=$(query_prometheus 'snmp_gauge{device_name="E2E-SIM",resolved_name="[NAME]"}')
    RESULT_COUNT=$(echo "$RESPONSE" | jq -r '.data.result | length')

    if [ "$RESULT_COUNT" -eq 0 ]; then
        record_fail "$SCENARIO_NAME" "no snmp_gauge series found for [NAME]"
    else
        VALUE=$(echo "$RESPONSE" | jq -r '.data.result[0].value[1]')
        VALUE_INT=$(echo "$VALUE" | cut -d. -f1)
        EVIDENCE="resolved_name=[NAME] value=${VALUE}"

        if [ "$VALUE_INT" = "[EXPECTED]" ]; then
            record_pass "$SCENARIO_NAME" "$EVIDENCE"
        else
            record_fail "$SCENARIO_NAME" "expected [EXPECTED] got ${VALUE_INT}. ${EVIDENCE}"
        fi
    fi
    ```

   ```csharp
   IPAddress ip;
   if (IPAddress.TryParse(d.IpAddress, out var parsed))

       ip = parsed.MapToIPv4();
   }

   {
       // Resolve K8s Service DNS name to IP at startup
       var addresses = System.Net.Dns.GetHostAddresses(d.IpAddress);
       ip = addresses.First(a => a.AddressFamily == System.Net.Sockets.AddressFamily.InterNetwork);
   }
   ```
